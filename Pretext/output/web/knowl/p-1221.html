On calcule \(a_n\) et \(b_n\) comme avant. Ensuite, on écrit les intégrales en changeant les variables de \(s\) à \(t\) tel que \(ds = \frac{\pi}{L} \, dt\text{.}\)
\begin{equation*} \mybxbg{~~ \begin{aligned} \amp a_0 = \frac{1}{\pi} \int_{-\pi}^\pi g(s) \, ds = \frac{1}{L} \int_{-L}^L f(t) \, dt , \\ \amp a_n = \frac{1}{\pi} \int_{-\pi}^\pi g(s) \, \cos (n s) \, ds = \frac{1}{L} \int_{-L}^L f(t) \, \cos \left( \frac{n \pi}{L} t \right) \, dt , \\ \amp b_n = \frac{1}{\pi} \int_{-\pi}^\pi g(s) \, \sin (n s) \, ds = \frac{1}{L} \int_{-L}^L f(t) \, \sin \left( \frac{n \pi}{L} t \right) \, dt . \end{aligned} ~~} \end{equation*}
