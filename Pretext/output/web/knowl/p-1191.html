\begin{equation*} \begin{split} b_n \amp = \frac{1}{\pi} \int_{-\pi}^\pi f(t) \sin (nt) \,dt \\ \amp = \frac{1}{\pi} \int_{0}^\pi \pi \sin (nt) \,dt \\ \amp = \left[ \frac{- \cos (nt)}{n} \right]_{t=0}^\pi \\ \amp = \frac{1 - \cos (\pi n)}{n} = \frac{1 - {(-1)}^n}{n} = \begin{cases} \frac{2}{n} \amp \text{ si } n \text{ est impair } , \\ 0 \amp \text{ si } n \text{ est pair } . \end{cases} \end{split} \end{equation*}
